    \begin{itemize}
        \item Rutovitz
            \[NC = \sum_{k=1}^8 \left| x_k - x_{k+1}\right|\]
        \item Yokoi (4-conexo)
            \[NC = \sum_{k \in V_4} \left| x_k - x_k \cdot x_{k+1} \cdot x_{k+2}\right|\]
        \item Yokoi (4-conexo)
        \[NC = \sum_{k \in V_4} \left| \overline{x_k} - \overline{x_k} \cdot \overline{x_{k+1}} \cdot \overline{x_{k+2}}\right|\]
    \end{itemize}
